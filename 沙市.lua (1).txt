local configPath = "BS脚本/UI配置.txt"
local colorConfigPath = "BS脚本/颜色配置.txt"

local function saveConfig(config)
	local success, encoded = pcall(function()
		return game:GetService("HttpService"):JSONEncode(config)
	end)
	if success then
		if isfolder("BS脚本") then
			if isfile(configPath) then
				delfile(configPath)
			end
		else
			makefolder("BS脚本")
		end
		writefile(configPath, encoded)
	end
end

local function loadConfig()
	if isfile(configPath) then
		local success, config = pcall(function()
			return game:GetService("HttpService"):JSONDecode(readfile(configPath))
		end)
		if success then
			return config
		end
	end
	return {
		autoFitScreen = false,
		windowSize = {650, 400},
		windowPosition = {0.5, 0.5}
	}
end

local function CreateGlassEffect(instance, transparency)
	instance.BackgroundTransparency = transparency or Colors.Transparent

function Tween(obj, duration, style, direction, data)
	services.TweenService
		:Create(obj, TweenInfo.new(duration, Enum.EasingStyle[style], Enum.EasingDirection[direction]), data)

local function autoFitUIScreen(mainFrame)
	local viewportSize = services.Workspace.CurrentCamera.ViewportSize
	
	if viewportSize.X < 800 then
		mainFrame.Size = UDim2.new(0, viewportSize.X - 100, 0, viewportSize.Y - 100)
	elseif viewportSize.Y < 600 then
		mainFrame.Size = UDim2.new(0, 550, 0, viewportSize.Y - 100)
	else
		mainFrame.Size = UDim2.new(0, 650, 0, 400)
	end
	
	mainFrame.Position = UDim2.new(0.5, 0, 0.5, 0)
end

function drag(frame, hold)
	if not hold then
		hold = frame
	end
	local dragging
	local dragInput
	local dragStart
	local startPos
	local function update(input)
		local delta = input.Position - dragStart
		frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
	hold.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging = true
			dragStart = input.Position
			startPos = frame.Position
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)
	frame.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement then
			dragInput = input
		end
	end)
	services.UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input)
		end
	end)
end

local function highlightText(originalText, searchTerm, color)
	if searchTerm == "" or not originalText:lower():find(searchTerm:lower(), 1, true) then
		return originalText
	end
	
	local pattern = searchTerm:gsub("([%^%$%(%)%%%.%[%]%*%+%-%?])", "%%%1")
	local highlighted = originalText:gsub("(" .. pattern .. ")", 
		'<font color="#4A90E2"><b>%1</b></font>')
	return highlighted
end

			new.frame.Visible = true

			Tween(new.button, 0.15, "Sine", "Out", { BackgroundTransparency = 0 })
			Tween(new.button.TabText, 0.15, "Sine", "Out", { TextTransparency = 0 })
			return
		end
		if old.button == new.button then return end
		
		switchingTabs = true
		library.currentTab = new
		
		Tween(old.button, 0.15, "Sine", "Out", { BackgroundTransparency = 0.3 })
		Tween(new.button, 0.15, "Sine", "Out", { BackgroundTransparency = 0 })
		Tween(old.button.TabText, 0.15, "Sine", "Out", { TextTransparency = 0.3 })
		Tween(new.button.TabText, 0.15, "Sine", "Out", { TextTransparency = 0 })
		
		old.frame.Visible = false
		new.frame.Visible = true
		
		task.wait(0.15)
		switchingTabs = false
	end

				local function updateColorPicker()
					local r, g, b = currentColor.r, currentColor.g, currentColor.b
					local max, min = math.max(r, g, b), math.min(r, g, b)
					val = max
					local delta = max - min
					
					if delta == 0 then
						hue, sat = 0, 0
					else
						sat = delta / max
						if r == max then
							hue = (g - b) / delta
						elseif g == max then
							hue = 2 + (b - r) / delta
						else
							hue = 4 + (r - g) / delta
						end
						hue = hue / 6
						if hue < 0 then hue = hue + 1 end
					end
					
					SatValPicker.BackgroundColor3 = Color3.fromHSV(hue, 1, 1)
					SatValCursor.Position = UDim2.new(sat, -8, 1 - val, -8)
					HueSlider.Position = UDim2.new(0, 0, hue, 0)
				end
				
				updateColorPicker()
				updateSavedColors()

					currentColor = Color3.fromHSV(hue, sat, val)
					ColorDisplay.BackgroundColor3 = currentColor
					library.flags[flag] = currentColor
					callback(currentColor)
					HueSlider.Position = UDim2.new(0, 0, hue, 0)
				end
				
				SatValPicker.InputBegan:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 then
						local conn
						conn = services.UserInputService.InputChanged:Connect(function(input)
							if input.UserInputType == Enum.UserInputType.MouseMovement then
								local pos = SatValPicker.AbsolutePosition
								updateSatVal(input.Position.X - pos.X, input.Position.Y - pos.Y)
							end
						end)
						
						services.UserInputService.InputEnded:Connect(function(input)
							if input.UserInputType == Enum.UserInputType.MouseButton1 then
								conn:Disconnect()
							end
						end)
						
						local pos = SatValPicker.AbsolutePosition
						updateSatVal(input.Position.X - pos.X, input.Position.Y - pos.Y)
					end
				end)
				
				HuePicker.InputBegan:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 then
						local conn
						conn = services.UserInputService.InputChanged:Connect(function(input)
							if input.UserInputType == Enum.UserInputType.MouseMovement then
								local pos = HuePicker.AbsolutePosition
								updateHue(input.Position.Y - pos.Y)
							end
						end)
						
						services.UserInputService.InputEnded:Connect(function(input)
							if input.UserInputType == Enum.UserInputType.MouseButton1 then
								conn:Disconnect()
							end
						end)
						
						local pos = HuePicker.AbsolutePosition
						updateHue(input.Position.Y - pos.Y)
					end
				end)
				
				SaveColorBtn.MouseButton1Click:Connect(function()
					local r, g, b = currentColor.r * 255, currentColor.g * 255, currentColor.b * 255
					local colorData = {math.floor(r), math.floor(g), math.floor(b)}
					
					local exists = false
					for _, savedColor in ipairs(colorConfig.savedColors) do
						if savedColor[1] == colorData[1] and savedColor[2] == colorData[2] and savedColor[3] == colorData[3] then
							exists = true
							break
						end
					end
					
					if not exists then
						table.insert(colorConfig.savedColors, 1, colorData)
						if #colorConfig.savedColors > 20 then
							table.remove(colorConfig.savedColors, 21)
						end
						saveColorConfig(colorConfig)
						updateSavedColors()
					end
				end)
				
				ColorPickerBtn.MouseButton1Click:Connect(function()
					ColorPickerWindow.Visible = not ColorPickerWindow.Visible
				end)
				
				services.UserInputService.InputBegan:Connect(function(input)
					if input.UserInputType == Enum.UserInputType.MouseButton1 then
						local mousePos = Vector2.new(mouse.X, mouse.Y)
						local windowPos = ColorPickerWindow.AbsolutePosition
						local windowSize = ColorPickerWindow.AbsoluteSize
						
						if not (mousePos.X >= windowPos.X and mousePos.X <= windowPos.X + windowSize.X and
								mousePos.Y >= windowPos.Y and mousePos.Y <= windowPos.Y + windowSize.Y) then
							local btnPos = ColorPickerBtn.AbsolutePosition
							local btnSize = ColorPickerBtn.AbsoluteSize
							if not (mousePos.X >= btnPos.X and mousePos.X <= btnPos.X + btnSize.X and
									mousePos.Y >= btnPos.Y and mousePos.Y <= btnPos.Y + btnSize.Y) then
								ColorPickerWindow.Visible = false
							end
						end
					end
				end)
				
				local funcs = {
					SetValue = function(self, color)
						if typeof(color) == "Color3" then
							currentColor = color
							ColorDisplay.BackgroundColor3 = currentColor
							library.flags[flag] = currentColor
							callback(currentColor)
							updateColorPicker()
						end
					end,
					GetValue = function()
						return currentColor
					end,
					Module = ColorPickerModule,
				}
				
				return funcs
			end
			
			section.TextBox = section.Textbox
			section.textbox = section.Textbox
			section.KeyBind = section.Keybind
			section.keybind = section.Keybind
			section.ColorPicker = section.ColorPicker
			section.colorpicker = section.ColorPicker
			
			return section
		end
		
		return tab
	end
	
	local SCALE = 0.01
	local LookView = Vector3.new(0, 0.5, -10)
	local dragOffset3D = Vector2.new(0, 0)
	local dragging3D = false
	local dragStart3D = Vector2.new(0, 0)

	TitleBar.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging3D = true
			dragStart3D = Vector2.new(input.Position.X, input.Position.Y)
			dragOffset3D = Vector2.new(LookView.X, LookView.Y)
		end
	end)

	TitleBar.InputEnded:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 then
			dragging3D = false
		end
	end)

	services.UserInputService.InputChanged:Connect(function(input)
		if dragging3D and input.UserInputType == Enum.UserInputType.MouseMovement then
			local delta = Vector2.new(input.Position.X - dragStart3D.X, input.Position.Y - dragStart3D.Y) * 0.01
			LookView = Vector3.new(dragOffset3D.X + delta.X, dragOffset3D.Y - delta.Y, LookView.Z)
		end
	end)

	game:GetService("RunService").RenderStepped:Connect(function(dt)
		if not canvasPart or not canvasPart.Parent then return end
		
		local offsetX = (mouse.X - mouse.ViewSizeX/2) * SCALE
		local offsetY = (mouse.Y - mouse.ViewSizeY/2) * SCALE
		
		local goalCFrame = camera.CFrame 
			* CFrame.new(LookView.X, LookView.Y, LookView.Z)
			* CFrame.Angles(0, math.rad(offsetX), 0)
			* CFrame.Angles(math.rad(offsetY), 0, 0)
		
		services.TweenService:Create(canvasPart, TweenInfo.new(dt * 2), {
			CFrame = goalCFrame
		}):Play()
	end)
	
	function window:Set3DOffset(offsetVector)
		LookView = offsetVector
	end

	function window:GetCanvasPart()
		return canvasPart
	end

	function window:Destroy()
		if canvasPart then
			canvasPart:Destroy()
		end
	end

	return window
end

return library
